**1.** (例 6.3.2)
$$
\env{aligned}{
	I &= \int \frac{x^3+1}{x^4-3x^3+3x^2-x}\d x\\
	&= \int \frac{x^3+1}{x(x-1)^3}\d x\\
	&= \int\br{\frac{A}{x}+\frac{B}{x-1}+\frac{C}{(x-1)^2}+\frac{D}{(x-1)^3}}\d x.
}
$$
其中, $A(x-1)^3+Bx(x-1)^2+Cx(x-1)+Dx=x^3+1$, 即
$$
\env{cases}{
	-A=1,\\
	3A+B-C+D=0,\\
	-3A-2B+C=0,\\
	A+B=1.
}\Ra\env{cases}{
	A=-1,\\
	B=2,\\
	C=1,\\
	D=2.
}
$$
所以
$$
\env{aligned}{
	I &= -\int\frac{\d x}{x}+2\int\frac{\d x}{x-1}+\int\frac{\d x}{(x-1)^2}+2\int\frac{\d x}{(x-1)^3}\\
	&= -\ln|x|+2\ln|x-1|-\frac{1}{x-1}-\frac{1}{(x-1)^2}+C.
}
$$

**2.** (例 6.3.6)
$$
\env{aligned}{
	I &= \int \frac{\cos^2 x}{(a^2\sin^2 x+b^2\cos^2 x)^2}\d x\\
	&= \int \frac{\cos^4 x}{(a^2\sin^2 x+b^2\cos^2 x)^2\cos^2 x}\d x\\
	&= \int \frac{1}{(a^2\tan^2 x+b^2)^2}\d\tan x\\
	&= \frac{1}{a^4}\int \frac{1}{\br{t^2+k^2}^2}\d t.\quad(t=\tan x,~k=b^2/a^2)\\
}
$$
&emsp;&emsp;回忆有, 对于以下积分:
$$
I_n=\int\frac{1}{(x^2+a^2)^n}\d x,
$$
给出递推公式
$$
\env{aligned}{
	I_n &= \frac{x}{(x^2+a^2)^n}-\int x\d\frac{1}{(x^2+a^2)^n}\\
	&= \frac{x}{(x^2+a^2)^n}+2n\int \frac{x^2}{(x^2+a^2)^{n+1}}\\
	&= \frac{x}{(x^2+a^2)^n}+2n(I_n-a^2I_{n+1}),
}
$$
即
$$
I_{n+1}=\frac{1}{2na^2}\br{\frac{x}{(x^2+a^2)^n}+(2n-1)I_n}.
$$
边界有
$$
I_1=\frac{1}{a}\int\frac{1}{\frac{x^2}{a^2}+1}\d\frac{x}{a}=\frac{1}{a}\arctan\frac{x}{a}+C.
$$
&emsp;&emsp;回到原题, 代入公式有
$$
I_1=\frac{1}{k}\arctan\frac{x}{k}+C,\\
I_2=\frac{1}{2k^2}\br{\frac{x}{x^2+k^2}+\frac{1}{k}\arctan\frac{x}{k}}+C.
$$
因此最终答案
$$
\env{aligned}{
	I &= \frac{1}{a^4}\cdot\frac{a^2}{2b^2}\br{\frac{a^2t}{a^2t^2+b^2}+\frac{a}{b}\arctan\frac{at}{b}}+C\\
	&= \frac{1}{2a^2b^2}\br{\frac{a^2\tan x}{a^2\tan^2 x+b^2}+\frac{a}{b}\arctan\br{\frac{a}{b}\tan x}}+C.
}
$$

\env{cases}{

因此
$$
\env{aligned}{

**5.** (习题六 7)

&emsp;&emsp;(10)
$$
\env{aligned}{
	I &= \int\frac{\sin x}{\sin x-\cos x+2}\d x\\
	&= \int\frac{\frac{2t}{1+t^2}}{\frac{2t}{1+t^2}-\frac{1-t^2}{1+t^2}+2}\frac{2}{1+t^2}\d t\\
	&= 4\int\frac{t\d t}{(1+t^2)(2t-(1-t^2)+2(1+t^2))}\\
	&= 4\int\frac{t\d t}{(1+t^2)(3t^2+2t+1)}\\
	&= 4\int\br{\frac{At+B}{1+t^2}+\frac{Ct+D}{3t^2+2t+1}}\d t.
}
$$

其中
$$
(At+B)(3t^2+2t+1)+(Ct+D)(1+t^2)=t\\
\Ra\env{cases}{
	3A+C=0,\\
	2A+3B+D=0,\\
	A+2B+C=1,\\
	B+D=0
}\Ra\env{cases}{
	A=-\frac{1}{4},\\
	B=\frac{1}{4},\\
	C=\frac{3}{4},\\
	D=-\frac{1}{4}.
}
$$
代回得
$$
\env{aligned}{
	I &= \int\frac{1-t}{1+t^2}\d t+\int\frac{3t-1}{3t^2+2t+1}\d t\\
	&= \arctan t-\frac{1}{2}\ln(1+t^2)+\int\frac{t-\frac{1}{3}}{\br{t+\frac{1}{3}}^2+\frac{2}{9}}\d t\\
	&= \arctan t-\frac{1}{2}\ln(1+t^2)+\frac{1}{2}\int\frac{\d\br{t+\frac{1}{3}}^2}{\br{t+\frac{1}{3}}^2+\frac{2}{9}}-3\int\frac{\d t}{\br{\frac{3t+1}{\sqrt 2}}^2+1}\\
	&= \arctan t-\frac{1}{2}\ln(1+t^2)+\frac{1}{2}\ln\abs{t^2+\frac{2}{3}t+\frac{1}{3}}-\sqrt 2\int\frac{\d\br{\frac{3t+1}{\sqrt 2}}}{\br{\frac{3t+1}{\sqrt 2}}^2+1}\\
	&= \frac{x}{2}-\frac{1}{2}\ln\br{1+\tan^2\frac{x}{2}}+\frac{1}{2}\ln\abs{\tan^2\frac{x}{2}+\frac{2}{3}\tan\frac{x}{2}+\frac{1}{3}}-\sqrt 2\arctan \frac{3\tan\frac{x}{2}+1}{\sqrt 2}+C.
}
$$
